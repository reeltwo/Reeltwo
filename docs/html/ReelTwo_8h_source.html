307constexpr uint16_t crc16_impl(const char* p, size_t len, uint16_t crc)
308{
309 return len ?
310 crc16_impl(p+1,len-1,(crc>>8)^_crc16_table[((crc^*p)&0xFF)])
311 : crc;
312}

314constexpr uint32_t crc32_impl(const char* p, size_t len, uint32_t crc)
315{
316 return len ?
317 crc32_impl(p+1L,len-1L,(crc>>8L)^_crc32_table[(crc&0xFF)^(*p&0xFF)])
318 : crc;
319}
