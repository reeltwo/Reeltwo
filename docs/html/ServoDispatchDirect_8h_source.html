197 virtual uint16_t scaleToPos(uint16_t num, float scale) override
198 {
199 uint16_t pos = 0;
200 if (num < numServos)
201 {
202 scale = min(max(0.0f, scale), 1.0f);
203 uint16_t startPulse = fServos[num].startPulse;
204 uint16_t endPulse = fServos[num].endPulse;
205 if (startPulse < endPulse)
206 {
207 pos = startPulse + (endPulse - startPulse) * scale;
208 }
209 else
210 {
211 pos = startPulse - (startPulse - endPulse) * scale;

365 void move(ServoDispatchDirect<numServos>* dispatch, uint32_t timeNow)
366 {
367 if (finishTime != 0)
368 {
369 if (timeNow < startTime)
370 {
371 /* wait */
372 }
373 else if (timeNow >= finishTime)
374 {
375 posNow = finishPos;
376 doMove(dispatch, timeNow);
377 init();
378 detachTime = timeNow + 500;
379 }
380 else if (lastMoveTime != timeNow)
381 {
382 uint32_t timeSinceLastMove = timeNow - startTime;
383 uint32_t denominator = finishTime - startTime;
384 float (*useMethod)(float) = easingMethod;
385 if (useMethod == nullptr)
386 useMethod = Easing::LinearInterpolation;
387 float fractionChange = useMethod(float(timeSinceLastMove)/float(denominator));
388 int distanceToMove = float(deltaPos) * fractionChange;
389 uint16_t newPos = startPosition + distanceToMove;
390 if (newPos != posNow)
391 {
392 posNow = startPosition + distanceToMove;
393 doMove(dispatch, timeNow);
394 }
395 }
